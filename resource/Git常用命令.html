Git常用命令

太阳丶 2021-11-9 Git

# Git 常用命令

# 全局配置

  • 检查配置信息（配置信息保存在.gitconfig文件下）

    git config --list // 列出所有 Git 当时能找到的配置
git config --global --list // 列出全局配置信息
git config --system --list // 列出系统配置信息

  • 修改用户信息

    git config --global user.name "John Doe" // 修改用户名
git config --global user.email johndoe@example.com // 修改用户邮箱

# 仓库管理

  • 基础命令

    git init // 初始化git
git status // 查看文件状态
git log // 查看提交日志
git add <paths> // 暂存文件
git commit -m <message> // 提交暂存文件，并添加描述

  • 远程仓库

    git clone <url> // 克隆远程仓库
git remote -v // 查看所有远程仓库
git remote add <shortname> <url> // 添加一个新的远程 Git 仓库，同时指定一个方便使用的简写
git remote rename <shortname> <newname> // 重命名远程仓库
git remote remove|rm <shortname> // 删除远程仓库
git remote set-url <shortname> <url> // 修改远程仓库url
git fetch <remote> // 从远程中拉取数据
git pull --rebase// 拉取本地分支对应的远程分支数据，并自动进行合并，git pull = git fetch + git merge,在实际使用中，git fetch更安全一些
git push --force orgin master // 强制覆盖远程仓库
git push --set-upstream|-u origin master // 推送并把远程仓库和本地仓库关联
上面命令可以用下面两条命令代替
git push origin master
git branch [--set-upstream-to=origin/master|-u origin/master] master

  • 标签管理

    git tag // 列出标签
git tag -a <tagname> -m <message> // 创建标注标签
git tag <tagname>  // 创建轻量标签
git tag -d <tagname> // 删除本地标签
git push origin <tagname> // 标签共享至远程
git push origin --tags // 共享所有标签到远程
git push origin :refs/tags/<tagname> // 删除远程标签第一种方法
git push origin --delete <tagname> // 删除远程标签第二种方式
git checkout <tagname> // 检出标签，注意检出的标签不能作为分支使用
git checkout -b <branch> <tagname> // 将标签检出为一个新的分支

  • 时光穿梭

    git commit --amend // 重新提交
git reset HEAD <paths> // 撤销暂存文件
git checkout -- <paths> // 撤销对文件的修改，Git会用最近的版本覆盖它
git revert <commit> //  撤销指定的版本，撤销也会作为一次提交进行保存
git reset --hard HEAD^ // 回退到上一版本
git reset --hard <commit> // 回退到指定版本
git reflog // 查看命令历史

  • 分支管理

    git branch -a // 列出本地和远程所有分支
git branch <branch> // 创建分支
git checkout|switch <branch> // 切换分支
git checkout -b <branch> [origin/master]? 或 git checkout --track origin/serverfix // 创建并切换分支或从远程拉取分支到本地
git branch -d|-D <branch> // 删除分支
git checkout <branch> <paths> // 从其他分支获取文件或文件夹
git push --delete origin <branch> // 跟删除远程标签一样删除远程分支
git merge <branch> // 合并分支
git merge (--continue | --abort | --quit) // （继续，撤销，静默）合并
git rebase <branch> // 获取目标分支变化，并进行时间线变基
git rebase (--continue | --abort | --quit) // （继续，撤销，静默）变基

  • 临时储存

    git stash list // 查看储存列表
git stash [save message] // 保存，save为可选项，message为本次储存的注释
git stash pop stash@{num} // 恢复，num是可选项，恢复完会删除储存
git stash apply stash@{num} // 恢复，num是可选项，恢复完不会删除储存
git stash drop stash@{num} // 删除某个储存
git stash clear // 删除所有储存
